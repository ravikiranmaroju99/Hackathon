cat > app/web/app.js <<'EOF'
const API_URL = "__API_URL__";
const STORAGE_KEY = "opsai-conversations-v1";

const messagesElement = document.getElementById("messages");
const conversationList = document.getElementById("conversation-list");
const form = document.getElementById("chat-form");
const input = document.getElementById("message-input");
const sendButton = document.getElementById("send-button");
const newChatButton = document.getElementById("new-chat-button");
const clearHistoryButton = document.getElementById("clear-history-button");
const sidebar = document.getElementById("sidebar");
const mobileMenuButton = document.getElementById("mobile-menu-button");

let conversations = loadConversations();
let activeConversationId = null;
let isSending = false;

function createId(prefix = "id") {
    return `${prefix}-${Date.now()}-${Math.random().toString(16).slice(2)}`;
}

function loadConversations() {
    try {
        const saved = localStorage.getItem(STORAGE_KEY);
        return saved ? JSON.parse(saved) : [];
    } catch (error) {
        console.error("Unable to read chat history:", error);
        return [];
    }
}

function saveConversations() {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(conversations));
}

function getActiveConversation() {
    return conversations.find(
        conversation => conversation.id === activeConversationId
    );
}

function createConversation() {
    const conversation = {
        id: createId("chat"),
        sessionId: createId("session"),
        title: "New conversation",
        messages: [],
        updatedAt: Date.now()
    };

    conversations.unshift(conversation);
    activeConversationId = conversation.id;
    saveConversations();
    renderAll();
}

function setConversationTitle(conversation, question) {
    if (conversation.messages.length === 1) {
        conversation.title =
            question.length > 42
                ? `${question.slice(0, 42)}...`
                : question;
    }
}

function renderConversationList() {
    conversationList.innerHTML = "";

    conversations.forEach(conversation => {
        const button = document.createElement("button");
        button.className = "conversation-item";
        button.textContent = conversation.title;

        if (conversation.id === activeConversationId) {
            button.classList.add("active");
        }

        button.addEventListener("click", () => {
            activeConversationId = conversation.id;
            renderAll();
            sidebar.classList.remove("open");
        });

        conversationList.appendChild(button);
    });
}

function renderWelcome() {
    messagesElement.innerHTML = `
        <div class="welcome-card">
            <div class="assistant-avatar large-avatar">O</div>
            <h2>Hello, I am OpsAI Assistant</h2>
            <p>
                Ask questions about Kubernetes, RKE2, Jenkins,
                cloud and infrastructure operations.
            </p>
            <div class="suggestion-grid">
                <button class="suggestion">
                    What should I check when a Kubernetes node becomes unavailable?
                </button>
                <button class="suggestion">
                    How do I troubleshoot Kubernetes DiskPressure?
                </button>
                <button class="suggestion">
                    What should I check when a Jenkins pipeline fails?
                </button>
            </div>
            <div class="approval-notice">
                Production changes, reboots and deletions require human approval.
            </div>
        </div>
    `;

    bindSuggestionButtons();
}

function addMessageElement(message) {
    const row = document.createElement("div");
    row.className = `message-row ${message.role}`;

    if (message.role === "assistant") {
        const avatar = document.createElement("div");
        avatar.className = "assistant-avatar";
        avatar.textContent = "O";
        row.appendChild(avatar);
    }

    const bubble = document.createElement("div");
    bubble.className = "message-bubble";

    const content = document.createElement("div");
    content.textContent = message.content;
    bubble.appendChild(content);

    const meta = document.createElement("div");
    meta.className = "message-meta";
    meta.textContent =
        message.role === "assistant"
            ? "OpsAI Assistant"
            : "You";

    bubble.appendChild(meta);
    row.appendChild(bubble);
    messagesElement.appendChild(row);
}

function renderMessages() {
    const conversation = getActiveConversation();

    if (!conversation || conversation.messages.length === 0) {
        renderWelcome();
        return;
    }

    messagesElement.innerHTML = "";
    conversation.messages.forEach(addMessageElement);
    scrollToBottom();
}

function renderAll() {
    renderConversationList();
    renderMessages();
}

function scrollToBottom() {
    messagesElement.scrollTop = messagesElement.scrollHeight;
}

function showTyping() {
    const row = document.createElement("div");
    row.id = "typing-row";
    row.className = "message-row assistant";

    row.innerHTML = `
        <div class="assistant-avatar">O</div>
        <div class="message-bubble typing">
            OpsAI Assistant is checking the approved runbooks...
        </div>
    `;

    messagesElement.appendChild(row);
    scrollToBottom();
}

function removeTyping() {
    document.getElementById("typing-row")?.remove();
}

async function sendMessage(question) {
    const cleanedQuestion = question.trim();

    if (!cleanedQuestion || isSending) {
        return;
    }

    if (!activeConversationId) {
        createConversation();
    }

    const conversation = getActiveConversation();

    conversation.messages.push({
        role: "user",
        content: cleanedQuestion
    });

    setConversationTitle(conversation, cleanedQuestion);
    conversation.updatedAt = Date.now();

    saveConversations();
    renderAll();

    input.value = "";
    resizeInput();

    isSending = true;
    sendButton.disabled = true;
    showTyping();

    try {
        const response = await fetch(API_URL, {
            method: "POST",
            headers: {
                "Content-Type": "application/json"
            },
            body: JSON.stringify({
                question: cleanedQuestion,
                sessionId: conversation.sessionId
            })
        });

        let result;

        try {
            result = await response.json();
        } catch {
            throw new Error("The backend returned an invalid response.");
        }

        if (!response.ok) {
            throw new Error(
                result.error || `Request failed with status ${response.status}`
            );
        }

        conversation.sessionId =
            result.sessionId || conversation.sessionId;

        conversation.messages.push({
            role: "assistant",
            content: result.answer || "No answer was returned."
        });
    } catch (error) {
        console.error(error);

        conversation.messages.push({
            role: "assistant",
            content:
                "I could not connect to the OpsAI backend. " +
                "Please try again or contact the platform administrator."
        });
    } finally {
        removeTyping();
        conversation.updatedAt = Date.now();
        saveConversations();

        isSending = false;
        sendButton.disabled = false;

        renderAll();
        input.focus();
    }
}

function bindSuggestionButtons() {
    document.querySelectorAll(".suggestion").forEach(button => {
        button.addEventListener("click", () => {
            sendMessage(button.textContent.trim());
        });
    });
}

function resizeInput() {
    input.style.height = "auto";
    input.style.height = `${Math.min(input.scrollHeight, 150)}px`;
}

form.addEventListener("submit", event => {
    event.preventDefault();
    sendMessage(input.value);
});

input.addEventListener("input", resizeInput);

input.addEventListener("keydown", event => {
    if (event.key === "Enter" && !event.shiftKey) {
        event.preventDefault();
        form.requestSubmit();
    }
});

newChatButton.addEventListener("click", () => {
    createConversation();
    sidebar.classList.remove("open");
    input.focus();
});

clearHistoryButton.addEventListener("click", () => {
    if (!window.confirm("Delete all saved chat history?")) {
        return;
    }

    conversations = [];
    activeConversationId = null;
    saveConversations();
    createConversation();
});

mobileMenuButton.addEventListener("click", () => {
    sidebar.classList.toggle("open");
});

if (conversations.length === 0) {
    createConversation();
} else {
    activeConversationId = conversations[0].id;
    renderAll();
}

input.focus();

# Find an existing Amplify application
AMPLIFY_APP_ID=$(aws amplify list-apps \
  --region "$AWS_REGION" \
  --query "apps[?name=='$AMPLIFY_APP_NAME'].appId | [0]" \
  --output text)

# Create the application when it does not exist
if [[ -z "$AMPLIFY_APP_ID" || "$AMPLIFY_APP_ID" == "None" ]]; then
    echo "Creating Amplify application..."

    AMPLIFY_APP_ID=$(aws amplify create-app \
      --name "$AMPLIFY_APP_NAME" \
      --description \
      "Browser interface for the OpsAI Assistant AWS MVP" \
      --platform WEB \
      --no-enable-branch-auto-build \
      --region "$AWS_REGION" \
      --query 'app.appId' \
      --output text)
else
    echo "Using existing Amplify application: $AMPLIFY_APP_ID"
fi

# Check whether the main branch exists
if ! aws amplify get-branch \
  --app-id "$AMPLIFY_APP_ID" \
  --branch-name "$AMPLIFY_BRANCH_NAME" \
  --region "$AWS_REGION" >/dev/null 2>&1
then
    echo "Creating Amplify main branch..."

    aws amplify create-branch \
      --app-id "$AMPLIFY_APP_ID" \
      --branch-name "$AMPLIFY_BRANCH_NAME" \
      --description "OpsAI MVP production branch" \
      --stage PRODUCTION \
      --framework "Web" \
      --no-enable-auto-build \
      --region "$AWS_REGION" \
      >/dev/null
else
    echo "Amplify main branch already exists."
fi

DEFAULT_DOMAIN=$(aws amplify get-app \
  --app-id "$AMPLIFY_APP_ID" \
  --region "$AWS_REGION" \
  --query 'app.defaultDomain' \
  --output text)

AMPLIFY_URL="https://${AMPLIFY_BRANCH_NAME}.${DEFAULT_DOMAIN}"
