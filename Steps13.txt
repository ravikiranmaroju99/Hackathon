Step 13B — Test the Deployed Agent from Python
Purpose

2. Create the Python test script
cat > app/test_agent.py <<'PY'
import os
import sys
import uuid

import boto3
from botocore.exceptions import BotoCoreError, ClientError


def main() -> int:
    region = os.environ.get("AWS_REGION", "us-east-1")
    agent_id = os.environ.get("BEDROCK_AGENT_ID")
    alias_id = os.environ.get("BEDROCK_AGENT_ALIAS_ID")

    if not agent_id or not alias_id:
        print("ERROR: Agent ID or Alias ID is missing from .env")
        return 1

    client = boto3.client(
        "bedrock-agent-runtime",
        region_name=region,
    )

    question = (
        "What should I check when a Kubernetes node becomes unavailable? "
        "Give me easy numbered steps."
    )

    print("Calling OpsAI Assistant...")
    print(f"Agent ID: {agent_id}")
    print(f"Alias ID: {alias_id}")
    print()

    try:
        response = client.invoke_agent(
            agentId=agent_id,
            agentAliasId=alias_id,
            sessionId=str(uuid.uuid4()),
            inputText=question,
            enableTrace=False,
        )

        answer_parts = []

        for event in response.get("completion", []):
            chunk = event.get("chunk")

            if chunk and "bytes" in chunk:
                answer_parts.append(
                    chunk["bytes"].decode("utf-8")
                )

        answer = "".join(answer_parts).strip()

        if not answer:
            print("No answer was received from the agent.")
            return 1

        print("OpsAI Assistant response:")
        print("-" * 60)
        print(answer)
        print("-" * 60)

        return 0

    except ClientError as error:
        print(f"AWS ERROR: {error}")
        return 1
    except BotoCoreError as error:
        print(f"SDK ERROR: {error}")
        return 1
    except Exception as error:
        print(f"UNEXPECTED ERROR: {error}")
        return 1


if __name__ == "__main__":
    sys.exit(main())
PY
